



## 编程题

1.HTML5用canvas绘制五星红旗

```html
<html>
  <head>
    <title>Canvas tutorial</title>
    <script type="text/javascript">
      function draw() {
		  var ctx = document.getElementById('canvas').getContext('2d');
		  //绘制红色的画布
		  ctx.fillStyle = '#f00';
		  ctx.fillRect(0,0,450,300);
		  ctx.translate(75,75);
		  //绘制大五角星 
		  ctx.save();
		  ctx.rotate(Math.PI*2/7);
		  drawStar(ctx,40);   
		  ctx.restore();
		  //绘制四个小五角星 
		  ctx.save();
		  ctx.translate(80,-50);
		  drawStar(ctx,10);   
		  ctx.restore();
		  ctx.save();
		  ctx.translate(110,-10);
		  drawStar(ctx,10);    
		  ctx.restore();
		  ctx.save();
		  ctx.translate(110,30);
		  drawStar(ctx,10);    
		  ctx.restore();
		  ctx.save();
		  ctx.translate(80,70);
		  drawStar(ctx,10);    
		  ctx.restore(); 
		}
		//绘制五角星函数
	function drawStar(ctx,r){
		  ctx.save();
		  ctx.beginPath()
		  ctx.moveTo(r,0);
		  for (var i=0;i<9;i++){
		    ctx.rotate(Math.PI/5);
		    if(i%2 == 0) {
		      ctx.lineTo((r/0.525731)*0.200811,0);
		    } else {
		      ctx.lineTo(r,0);
		    }
		  }
		  ctx.closePath();
		  ctx.fillStyle='#FFFF00';
		  ctx.fill();
		  ctx.restore();
	}
    </script>
  </head>
  <body onload="draw();">
    <canvas id="canvas" width="800" height="300"></canvas>
  </body>
</html>
```
